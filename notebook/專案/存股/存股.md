-- 屬性主表：定義你的技術棧
CREATE TABLE tech_stacks (
    stack_id SERIAL PRIMARY KEY,
    name VARCHAR(50), -- Architecture, Refactoring, Performance
    description TEXT,
    total_xp INTEGER DEFAULT 0,
    current_rank VARCHAR(20) DEFAULT 'Junior' -- Junior, Senior, Staff, Principal
);

-- 投資配置表：設定每月固定執行的任務
CREATE TABLE investment_tasks (
    task_id SERIAL PRIMARY KEY,
    stock_code VARCHAR(10), -- 例如 2330
    buy_day_of_month INTEGER, -- 1-31
    target_stack_id INTEGER REFERENCES tech_stacks(stack_id),
    amount_per_task DECIMAL(10, 2),
    is_enabled BOOLEAN DEFAULT TRUE
);


-- 1. 技術棧定義 (職人屬性)
CREATE TABLE tech_stacks (
    id SERIAL PRIMARY KEY,
    name VARCHAR(50) NOT NULL, -- Architecture, Performance, Refactoring
    current_rank VARCHAR(30) DEFAULT 'Junior Intern',
    total_loc_count INTEGER DEFAULT 0 -- 累計總行數 (買入次數或加權值)
);

-- 2. 投資 Commit 紀錄 (熱圖來源)
CREATE TABLE investment_commits (
    id SERIAL PRIMARY KEY,
    stack_id INTEGER REFERENCES tech_stacks(id),
    commit_date DATE DEFAULT CURRENT_DATE, -- 用來畫熱圖的 X 軸
    loc_added INTEGER NOT NULL, -- 本次買入貢獻的行數
    commit_message TEXT, -- 例如: "Fix: Strengthening the defensive line with 00878"
    is_recurring BOOLEAN DEFAULT TRUE -- 是否為每月固定扣款
);

-- 3. 等級門檻表 (定義什麼時候升職)
CREATE TABLE rank_thresholds (
    rank_name VARCHAR(30) PRIMARY KEY,
    required_loc INTEGER NOT NULL
);
