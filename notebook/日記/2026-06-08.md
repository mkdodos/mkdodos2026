庫存表點選明細,查看該筆料件的明細(切割歷程)

```sql
-- 查某一筆的往上所有層級

WITH RECURSIVE stock_tree AS (
  SELECT id, parent_id, 0 AS level
  FROM wp_stock
  WHERE id = 382          -- ← 從這筆往上找 

  UNION ALL  

  SELECT s.id, s.parent_id, st.level + 1
  FROM wp_stock s
  JOIN stock_tree st ON s.id = st.parent_id  -- ← 反向關聯
)

SELECT * FROM stock_tree;
```
