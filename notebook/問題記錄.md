在 Vite 設定的 proxy 功能，是 Vite 內建的開發伺服器（Dev Server）的反向代理（Reverse Proxy）。

當把專案打包（npm run build）並部署到 Nginx 時，Vite 的開發伺服器就不存在了。因此，必須在 Nginx 的設定檔（通常是 nginx.conf 或 default.conf）中，手動設定相同的反向代理規則。

```json
// vite.config.js
 server: {
    host: "0.0.0.0", // 或者直接寫 true
    proxy: {
      // 當你的前端程式碼呼叫 /api 開頭的路徑時
      "/api": {
        target: "http://192.168.0.10:3000", // 目標後端伺服器 IP
        changeOrigin: true, // 允許跨域
        // 如果你的後端 API 本身沒有 /api 前綴，可以透過 rewrite 拿掉它
        // rewrite: (path) => path.replace(/^\/api/, '')
      },
    },
  },
```

```json
// nginx.conf
server {
        listen       8080;
        server_name  localhost;

        # 1. 優先處理 API 請求（避免被下面的 location / 攔截）
        location /api/ {
            # 目標後端伺服器 IP 與 Port
            proxy_pass http://192.168.0.10:3000;             
            # 相當於 Vite 中的 changeOrigin: true
            proxy_set_header Host $host;
            # 傳遞真實的用戶 IP 給後端（好習慣，建議加上）
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }

        # 2. 處理前端靜態檔案
        location / {
            # 打包後的 dist 資料夾路徑
            root   D:\Dev\mkdodos2026\echoway\frontend\dist;
            index  index.html;
            # 支援 React Router 的 SPA 路由
            try_files $uri $uri/ /index.html;
        }
    }
```
